phpserialize-rs

High-performance PHP serialize/unserialize parser written in Rust with Python bindings. 10-100x faster than alternatives.

Get Started View on GitHub

Features

Zero-Copy Parsing

Minimal memory allocations using Rust's Cow (Copy-on-Write) for maximum performance. Parse gigabytes of data efficiently.

Full PHP Support

All PHP serialize types including objects, references, private/protected properties, and PHP 8.1+ enums.

UTF-8 Aware

Proper handling of multi-byte characters (Korean, Chinese, Japanese, etc.) with byte-length counting.

Auto DB Unescape

Automatic detection and handling of double-quoted strings from database exports.

PySpark Ready

Built-in UDFs for Databricks and Spark workloads. Both row-based and Pandas UDFs included.

Error Recovery

Configurable error handling with detailed position tracking for debugging malformed data.

Quick Start

from php_deserialize import loads, loads_json

# Basic usage
data = b'a:2:{s:4:"name";s:5:"Alice";s:3:"age";i:30;}'
result = loads(data)
print(result)  # {'name': 'Alice', 'age': 30}

# Direct JSON conversion (optimized for Databricks)
json_str = loads_json(data)
print(json_str)  # {"name":"Alice","age":30}

# Handle DB-escaped strings automatically
escaped = b'"a:1:{s:4:""key"";s:5:""value"";}"'
result = loads(escaped)  # Auto-unescapes
print(result)  # {'key': 'value'}

# Korean/UTF-8 support
korean = b's:6:"\xed\x95\x9c\xea\xb8\x80";'
print(loads(korean))  # '한글'
use php_deserialize_core::{from_bytes, PhpValue};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let data = br#"a:2:{s:4:"name";s:5:"Alice";s:3:"age";i:30;}"#;
    let value = from_bytes(data)?;

    match value {
        PhpValue::Array(items) => {
            for (key, val) in items {
                println!("{:?} => {:?}", key, val);
            }
        }
        _ => {}
    }

    Ok(())
}
from php_deserialize.spark import php_deserialize_udf, php_deserialize_pandas_udf

# Row-based UDF (simple)
deserialize = php_deserialize_udf("json")
df = df.withColumn("parsed", deserialize("php_data"))

# Pandas UDF (batch optimized, recommended for large datasets)
pandas_udf = php_deserialize_pandas_udf()
df = df.withColumn("parsed", pandas_udf("php_data"))

# With error handling
deserialize = php_deserialize_udf("json", errors="replace")
df = df.withColumn("parsed", deserialize("php_data"))

Error Handling

Basic Error Handling

from php_deserialize import loads, PhpDeserializeError

try:
    result = loads(b"invalid data")
except PhpDeserializeError as e:
    print(f"Error: {e}")
    # Error: unknown type marker 'i' at position 0

UTF-8 Error Modes

# Mode: "strict" - Raise on invalid UTF-8
loads(data, errors="strict")

# Mode: "replace" - Replace with (default)
loads(data, errors="replace")

# Mode: "bytes" - Return bytes for invalid UTF-8
loads(data, errors="bytes")

Validation Before Parsing

from php_deserialize import is_serialized, loads

data = get_data_from_somewhere()

if is_serialized(data):
    result = loads(data)
else:
    result = None  # Not PHP serialized

DB Escape Handling

# Auto-unescape (default)
loads(escaped_data, auto_unescape=True)

# Manual preprocessing
from php_deserialize import preprocess
clean_data = preprocess(escaped_data)
result = loads(clean_data, auto_unescape=False)
